JsonDic<JDFILENAME> _m_FILENAME_dic;
public void FILENAME_Pre()
{
	if(_m_FILENAME_dic==null)
	{
		if(m_FILENAME_dic.Count>0)return;
	}
}

public List<JDFILENAME> m_FILENAME_list
{
	get
	{
		return m_FILENAME_dic.DataList;
	}
}

public JsonDic<JDFILENAME> m_FILENAME_dic
{
	get
	{
		if(_m_FILENAME_dic==null)
		{
			string _data=LoadFile("FILENAME",true);
			if(string.IsNullOrEmpty(_data))
			{
				return null;
			}
			_m_FILENAME_dic=new JsonDic<JDFILENAME>(_data);
		}
		return _m_FILENAME_dic;
	}
}
