public JsonDic<JDvip> m_vip_dic
{
	get
	{
		if(_m_vip_dic==null)
		{
			string _data=LoadFile("vip",true);
			if(string.IsNullOrEmpty(_data))
			{
				return null;
			}
			_m_vip_dic=new JsonDic<JDvip>(_data);
		}
		return _m_vip_dic;
	}
}
