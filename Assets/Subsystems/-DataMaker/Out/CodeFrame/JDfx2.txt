public JsonDic<JDfx2> m_fx2_dic
{
	get
	{
		if(_m_fx2_dic==null)
		{
			string _data=LoadFile("fx2",true);
			if(string.IsNullOrEmpty(_data))
			{
				return null;
			}
			_m_fx2_dic=new JsonDic<JDfx2>(_data);
		}
		return _m_fx2_dic;
	}
}
