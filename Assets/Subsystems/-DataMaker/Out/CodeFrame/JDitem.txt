public JsonDic<JDitem> m_item_dic
{
	get
	{
		if(_m_item_dic==null)
		{
			string _data=LoadFile("item",true);
			if(string.IsNullOrEmpty(_data))
			{
				return null;
			}
			_m_item_dic=new JsonDic<JDitem>(_data);
		}
		return _m_item_dic;
	}
}
